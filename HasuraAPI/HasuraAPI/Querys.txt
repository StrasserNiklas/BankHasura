mutation UpdatePayment($id: Int!, $status: String) {
  update_payments_by_pk(
    pk_columns: {id: $id}
    _set: { status: $status}
  ) {
    status
  }
}

{
    "id": 5,
    "status": "Done"
}

query GetPayments($id: Int!) {
  payments(where: { sender_id: {_eq: $id}} ) {
    created_at
    amount
    status
    description
    sender {
      name
    }
    recipient {
      name
    }
  }
}

    "id": 1
}
